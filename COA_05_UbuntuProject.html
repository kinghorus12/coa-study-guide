Phase 1: Install & Verify MicroStack
Domain covered: Environment setup · ~30 minutes
Step 1.1 — System Preparation
# Update system
sudo apt update && sudo apt upgrade -y

# Install required packages
sudo apt install -y snapd git curl wget tree

# Verify virtualisation support
egrep -c "(vmx|svm)" /proc/cpuinfo
# Must return > 0. If 0, check BIOS VT-x setting.

# Check available disk
df -h /
Step 1.2 — Install MicroStack
# Install MicroStack snap
sudo snap install microstack --channel 2024/stable

# Initialise (takes 10-20 minutes — this is normal)
sudo microstack init --auto --control

# Get admin password
sudo snap get microstack config.credentials.keystone-password

# Source credentials
source /var/snap/microstack/common/etc/microstack.rc

# Verify ALL services running
openstack service list
openstack compute service list --long
openstack network agent list

# Expected output: nova-compute, nova-scheduler, nova-conductor,
# neutron-dhcp-agent, neutron-l3-agent, neutron-openvswitch-agent
Step 1.3 — Upload Base Image
# Download CirrOS (tiny test image — 16MB)
wget http://download.cirros-cloud.net/0.6.2/cirros-0.6.2-x86_64-disk.img

# Upload as our base image
openstack image create cloudcorelab-base \
  --disk-format qcow2 \
  --container-format bare \
  --file cirros-0.6.2-x86_64-disk.img \
  --public \
  --property project=cloudcorelab \
  --property os_type=linux \
  --tag production

# Verify
openstack image show cloudcorelab-base -c status -c visibility -c size

# DELIVERABLE: Take a screenshot of image list output
openstack image list
✅
Phase 1 Complete When: All OpenStack services are healthy, base image is uploaded and ACTIVE.

Step 2.1 — Create Domain, Projects, Users
# Create the project domain
openstack domain create cloudcorelab \
  --description "CloudCoreLab private cloud domain"

# Create all project tiers

  openstack project create ccl-$project \
    --domain cloudcorelab \
    --description "CloudCoreLab $project project"
done

# Create users
openstack user create ccl-webadmin --domain cloudcorelab --password WebAdmin2024!
openstack user create ccl-apidev   --domain cloudcorelab --password ApiDev2024!
openstack user create ccl-dba      --domain cloudcorelab --password DbAdmin2024!
openstack user create ccl-ops      --domain cloudcorelab --password Ops2024!

# Verify
openstack user list --domain cloudcorelab
Step 2.2 — Groups and Role Assignments
# Create functional groups
openstack group create ccl-web-team    --domain cloudcorelab
openstack group create ccl-api-team    --domain cloudcorelab

# Add users to groups
openstack group add user ccl-web-team ccl-webadmin --group-domain cloudcorelab --user-domain cloudcorelab
openstack group add user ccl-api-team ccl-apidev   --group-domain cloudcorelab --user-domain cloudcorelab

openstack security group create ccl-db-sg --description "DB tier: 5432 from API tier only"
openstack security group rule create ccl-db-sg --protocol tcp --dst-port 5432  --remote-ip 10.10.2.0/24

Step 4.2 — Web Tier Instances with Anti-Affinity
# Server group for web tier HA
openstack server group create ccl-web-group --policy anti-affinity
WEB_GROUP=$(openstack server group show ccl-web-group -f value -c id)

# Launch web servers with user data (auto-starts nginx)
cat > /tmp/web-init.sh << 'USERDATA'
#!/bin/bash
apt-get update -y
apt-get install -y nginx
systemctl enable nginx && systemctl start nginx
echo "CloudCoreLab Web Server" > /var/www/html/index.html
USERDATA

openstack server create ccl-web-01 \
  --image cloudcorelab-base \
  --flavor ccl.small \
  --network ccl-frontend-net \
  --security-group ccl-web-sg \
  --key-name ccl-deploy-key \
  --hint group=$WEB_GROUP \
  --user-data /tmp/web-init.sh \
  --wait

openstack server create ccl-web-02 \
  --image cloudcorelab-base \
  --flavor ccl.small \
  --network ccl-frontend-net \
  --security-group ccl-web-sg \
  --key-name ccl-deploy-key \
  --hint group=$WEB_GROUP \
  --user-data /tmp/web-init.sh \
  --wait

# Assign floating IPs to web tier
FIP1=$(openstack floating ip create ccl-public -f value -c floating_ip_address)
FIP2=$(openstack floating ip create ccl-public -f value -c floating_ip_address)
openstack server add floating ip ccl-web-01 $FIP1
openstack server add floating ip ccl-web-02 $FIP2

echo "Web tier floating IPs: $FIP1 $FIP2"

Step 6.1 — The Master Heat Template
This template reproduces the entire CloudCoreLab multi-tier environment as code. Writing this from scratch is the ultimate exam preparation exercise.
# File: /tmp/cloudcorelab-stack.yaml
heat_template_version: 2021-04-16

description: >
  CloudCoreLab full multi-tier private cloud stack.
  Deploys frontend and backend networks, security groups,
  and compute instances with floating IP on web tier.

parameters:
  base_image:
    type: string
    description: Base image for all instances
    default: cloudcorelab-base
  web_flavor:
    type: string
    description: Flavor for web tier
    default: ccl.small
  api_flavor:
    type: string
    description: Flavor for API tier
    default: ccl.medium
  keypair_name:
    type: string
    description: Keypair for SSH access
    default: ccl-deploy-key
  public_network:
    type: string
    description: External network name
    default: ccl-public
  dns_server:
    type: string
    default: 8.8.8.8

resources:

  # ── NETWORKS ─────────────────────────────────────────
  frontend_network:
    type: OS::Neutron::Net
    properties:
      name: stack-frontend-net

  frontend_subnet:
    type: OS::Neutron::Subnet
    properties:
      network: { get_resource: frontend_network }
      cidr: 10.20.1.0/24
      gateway_ip: 10.20.1.1
      dns_nameservers: [{ get_param: dns_server }]
      enable_dhcp: true

  backend_network:
    type: OS::Neutron::Net
    properties:
      name: stack-backend-net

  backend_subnet:
    type: OS::Neutron::Subnet
    properties:
      network: { get_resource: backend_network }
      cidr: 10.20.2.0/24
      gateway_ip: 10.20.2.1
      dns_nameservers: [{ get_param: dns_server }]

  # ── ROUTER ─────────────────────────────────────────
  core_router:
    type: OS::Neutron::Router
    properties:
      name: stack-core-router
      external_gateway_info:
        network: { get_param: public_network }

  frontend_router_iface:
    type: OS::Neutron::RouterInterface
    properties:
      router: { get_resource: core_router }
      subnet: { get_resource: frontend_subnet }

  backend_router_iface:
    type: OS::Neutron::RouterInterface
    properties:
      router: { get_resource: core_router }
      subnet: { get_resource: backend_subnet }

  # ── SECURITY GROUPS ─────────────────────────────────
  web_security_group:
    type: OS::Neutron::SecurityGroup
    properties:
      name: stack-web-sg
      rules:
        - protocol: icmp
        - protocol: tcp
          port_range_min: 22
          port_range_max: 22
        - protocol: tcp
          port_range_min: 80
          port_range_max: 80
        - protocol: tcp
          port_range_min: 443
          port_range_max: 443

  api_security_group:
    type: OS::Neutron::SecurityGroup
    properties:
      name: stack-api-sg
      rules:
        - protocol: tcp
          port_range_min: 8080
          port_range_max: 8080
          remote_ip_prefix: 10.20.1.0/24
        - protocol: tcp
          port_range_min: 22
          port_range_max: 22
          remote_ip_prefix: 10.20.1.0/24

  # ── COMPUTE ─────────────────────────────────────────
  web_server:
    type: OS::Nova::Server
    properties:
      name: stack-web-01
      image: { get_param: base_image }
      flavor: { get_param: web_flavor }
      key_name: { get_param: keypair_name }
      networks:
        - network: { get_resource: frontend_network }
      security_groups:
        - { get_resource: web_security_group }

  api_server:
    type: OS::Nova::Server
    properties:
      name: stack-api-01
      image: { get_param: base_image }
      flavor: { get_param: api_flavor }
      key_name: { get_param: keypair_name }
      networks:
        - network: { get_resource: backend_network }
      security_groups:
        - { get_resource: api_security_group }

  # ── FLOATING IP (web tier only) ──────────────────────
  web_floating_ip:
    type: OS::Neutron::FloatingIP
    properties:
      floating_network: { get_param: public_network }
      port_id: { get_attr: [web_server, addresses, stack-frontend-net, 0, port] }

  # ── STORAGE ─────────────────────────────────────────
  api_data_volume:
    type: OS::Cinder::Volume
    properties:
      name: stack-api-data
      size: 20

  api_volume_attach:
    type: OS::Cinder::VolumeAttachment
    properties:
      volume_id: { get_resource: api_data_volume }
      instance_uuid: { get_resource: api_server }

outputs:
  web_public_ip:
    description: Web server floating IP
    value: { get_attr: [web_floating_ip, floating_ip_address] }
  web_private_ip:
    description: Web server private IP
    value: { get_attr: [web_server, first_address] }
  api_private_ip:
    description: API server private IP
    value: { get_attr: [api_server, first_address] }
  frontend_network_id:
    description: Frontend network ID
    value: { get_attr: [frontend_network, id] }
  stack_summary:
    description: Stack deployment summary
    value:
      str_replace:
        template: "Web: WEB_IP | API: API_IP | Router: ROUTER_ID"
        params:
          WEB_IP:   { get_attr: [web_floating_ip, floating_ip_address] }
          API_IP:   { get_attr: [api_server, first_address] }
          ROUTER_ID: { get_resource: core_router }

# CloudCoreLab — Private Cloud Infrastructure

A production-equivalent multi-tier private cloud deployed on Ubuntu Server
using Canonical's MicroStack (OpenStack). Built as preparation for the
Certified OpenStack Administrator (COA) exam.

## Architecture

- **Platform**: MicroStack (Canonical snap-based OpenStack)
- **OS**: Ubuntu 24.04 LTS
- **OpenStack services**: Keystone, Nova, Neutron, Cinder, Swift, Heat, Glance

## What This Deploys

| Tier      | Network          | Instances         | Access      |
|-----------|------------------|-------------------|-------------|
| Web       | 10.10.1.0/24     | web-01, web-02    | Public FIP  |
| API       | 10.10.2.0/24     | api-server-01     | Internal    |
| Storage   | —                | 50GB Cinder vol   | Attached    |

## Infrastructure as Code

The entire environment is reproducible via a single Heat template:
\`\`\`bash
openstack stack create mycloud -t heat-templates/cloudcorelab-stack.yaml --wait
\`\`\`

## Skills Demonstrated

- OpenStack identity management (domains, projects, users, groups, RBAC)
- Multi-tier network design with tenant isolation and security group rules
- Nova compute lifecycle (launch, snapshot, resize, user-data, server groups)
- Cinder block storage (volume attach, extend, snapshot, restore)
- Swift object storage (containers, ACLs, metadata, auto-expire)
- Heat orchestration (HOT templates, intrinsic functions, stack lifecycle)
setup.sh — Automate the Full Deployment
#!/bin/bash
# CloudCoreLab full setup script
set -e

echo "=== CloudCoreLab Setup ==="

source /var/snap/microstack/common/etc/microstack.rc

echo "[1/6] Uploading base image..."
openstack image create cloudcorelab-base \
  --disk-format qcow2 --container-format bare \
  --file cirros-0.6.2-x86_64-disk.img --public 2>/dev/null || echo "Image exists"

echo "[2/6] Creating flavors..."
openstack flavor create ccl.small  --vcpus 1 --ram 1024 --disk 10 2>/dev/null || true
